Now let us show you how to create your game with us

#1 - Create Viewport and Scene

var viewport = new Engine.Viewport({
  id: 'viewport',
  background: 'black'
});

var camera = new Engine.Camera({
  scrollable: true
});

viewport.addCamera(camera);

var scene = new Engine.Scene();

camera.lookAt(scene);

#2 - Create Root Node

var layout = new Engine.Node();

scene.appendChild(layout);

#3 - Draw Borders

var AREA = {
  left: -400,
  top: -300,
  width: 800,
  height: 600
};

var BORDER = 30;

var top_bar = new Engine.Geometry.Rectangle({
  parent: layout,
  left: AREA.left,
  top: AREA.top,
  width: AREA.width,
  height: BORDER,
  fill: 'image(assets/images/bricks.png)'
});

// Same for left_bar and right_bar

#4 - Create Paddle and Ball

// CREATE PADDLE
var paddle = new Engine.Geometry.Rectangle({
  parent: layout,
  fill: 'image(assets/images/paddle.png)',
  width: 120,
  height: 20
});

// CREATE A BALL
var ball = new Engine.Geometry.Oval({
  parent: layout,
  fill: 'green',
  radius: 11
});

#4 - Initialize it

// SET PADDLE POSITION
var padd_left = - paddle.width / 2 + AREA.left + AREA.width / 2;
var paddle_top = AREA.top + AREA.height - paddle.height;

paddle.setPosition(padd_left, paddle_top);

// SET BALL POSITION
var ball_left = paddle.left + paddle.width / 2 - ball.width / 2;
var ball_top = paddle.top - ball.height;

ball.setPosition(ball_left, ball_top);

#5 - Prepare Bricks

var BRICKS_X = 12;
var BRICKS_Y = 5;
var BRICKS_MARGIN = 25;

var BRICK_WIDTH =  (AREA.width - 2 * BORDER - (BRICKS_X + 1)
           * BRICKS_MARGIN) / BRICKS_X;
var BRICK_HEIGHT = (AREA.height / 2 - 2 * BORDER - (BRICKS_Y + 1)
           * BRICKS_MARGIN) / BRICKS_Y;

// list of bricks instances, we'll fill this in the next step
var bricks = [];

function randomColor(){

  return ['red', 'orange','yellow'].random();

}

#6 - Initialize Bricks

function initBricks(){

  while(bricks.length) bricks.pop().destroy();

  for (var i = 0; i < BRICKS_X; i++){

    for (var j = 0; j < BRICKS_Y; j++) {

      var left = 2 * BORDER + AREA.left + i * BRICKS_MARGIN + i
              * BRICK_WIDTH;
      var top = 2 * BORDER + AREA.top + j * BRICKS_MARGIN + j
              * BRICK_HEIGHT;

      bricks.push(new Engine.Geometry.Rectangle({
        parent: layout,
        fill: randomColor(),
        left: left,
        top: top,
        width: BRICK_WIDTH,
        height: BRICK_HEIGHT
      }));

    }
  }
}

#7 - Initialize Game Loop

var game = new Engine.Timer({
  type: Engine.Timer.VSYNC,
  loop: true
});

#8 - Handling Input

var PADDLE_SPEED = 0,
  PADDLE_ACCELERATION = 0.4;

// key pressed
Engine.Input.on('keydown', function(e){
  switch(e.key){
    case 'A':
      PADDLE_SPEED -= PADDLE_ACCELERATION; break;
    case 'D':
      PADDLE_SPEED += PADDLE_ACCELERATION; break;
  }

// key release
}).on('keyup', function (e){
  switch(e.key){
    case 'A':
       PADDLE_SPEED += PADDLE_ACCELERATION; break;
    case 'D':
       PADDLE_SPEED -= PADDLE_ACCELERATION; break;
  }
});

#8 - Handling Input - update paddle

// update paddle position
game.on('step', function(){
  var dt = this.stepprogress;
  var left = Math.max(left_bar.right,
          Math.min(paddle.left + dt * PADDLE_SPEED,
            right_bar.left - paddle.width)
        );
  paddle.setPosition(left, paddle.top);
});

#9 - Moving the Ball

var BALL_ACCELERATION = 0.25,
  BALL_SPEED_X = BALL_ACCELERATION * [-1, 1].random(),
  BALL_SPEED_Y = 0;

game.on('step', function(){

  var dt = this.stepprogress;

  ball.setPosition(ball.left + dt * BALL_SPEED_X,
           ball.top + dt * BALL_SPEED_Y);

});

#10 - Detect Collisions

function overlap(a, b){

  return Engine.Box.overlap(a.left, a.top, a.width, a.height,
                b.left, b.top, b.width, b.height)

}

#10 - Detect Collisions

game.on('step', function(){

  if(overlap(ball, top_bar) || overlap(ball, paddle)){
    BALL_SPEED_Y *= -1;
    game.trigger('bounce');
  }

  if(overlap(ball, left_bar) || overlap(ball, right_bar)){
    BALL_SPEED_X *= -1;
    game.trigger('bounce');
  }

  if(!overlap(ball, AREA))
    game.trigger('over');
})

#11 - Handle Collisions

function collide(brick){

  if (overlap(ball, brick)){

    game.trigger('hit', brick);

    bricks.remove(brick);
    brick.destroy();

    if (!bricks.length)

      game.trigger('success');

    if (Math.random() < 0.5)

      BALL_SPEED_X *= -1;

    else

      BALL_SPEED_Y *= -1;

    return true;

  }

  return false;

}

#11 - Handle Collisions

// COLLIDE BALL WITH ANY OF BRICKS
game.on('step', function(){
  bricks.some(collide);
});

#12 - Game-over and start game

game.on('over', function(){

  game.stop();

  setTimeout(initGame, 3000);

});

// initialize game
function initGame(){

  initBricks();
  initPaddle();
  initBall();

  game.play();

}

initGame();

#13 - Sounds

var sounds = {
  collision: new Engine.Sound({

  }),

     src: "assets/sounds/bounce"
  })

}).on('hit', function(){

  sounds.collision.play();

});

#14 - Particles

game.on('hit', function(brick){

  // size of emitter geometry
  var width = 4 * brick.width;
  var height = 4 * brick.height;

  new Engine.Particles({
    parent: layout,
    iterations: 5,
    emitDelay: 50,
    redrawDelay: 1,
    width: width,
    height: height,
    left: brick.left + brick.width / 2 - width / 2,
    top: brick.top + brick.height / 2 - height / 2,
    amount: 15,
    on: {
      beforecreateparticle: function(i){

        this.lifetime = Math.random() * 500 + 900;
        this.originX = Math.random() * 20 - 10 + width / 2;
        this.originY = Math.random() * 20 - 10 + height / 2;
        this.dx = (Math.random() - 0.5)  * width;
        this.dy = (Math.random() - 0.5) * height;
        this.fill = brick.color + ' 5px 5px';
        this.opacity = Math.random() / 2;
      }
    }
  });
});
